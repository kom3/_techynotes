Input Validators in Javascript

function AllowAlphaNumeric(value) {
return value.replace(/[^A-Za-z0-9]/g, "");
}

function AllowAlphaNumericWithSpace(value) {
return value.replace(/[^A-Za-z0-9 ]/g, "");
}

function AllowOnlyNumeric(value) {
return value.replace(/[^0-9]/g, "");
}


function AllowAlphaWithSpace(value) {
return value.replace(/[^A-Za-z ]/g, "");
}
