Techynotes Docker Setup
Step: 1  Required files are
1. Dokerfile
2. requirements.txt file
3. git cloned techynotes source code
Step: 2 Dockerfile

From ubuntu
RUN apt-get update
WORKDIR /usr/src/
RUN apt-get install -y git
RUN apt-get install -y libsasl2-dev python-dev libldap2-dev libssl-dev psmisc
ADD techreq.txt .
RUN apt-get -y install python3-pip
RUN pip3 install -r techreq.txt
#RUN git clone -b feature/WAR-2339 https://github.com/warriorframework/warriorframework_py3.git
ADD ./techynotes ./techynotes
WORKDIR techynotes
RUN sed -i 's/ALLOWED_HOSTS.*/ALLOWED_HOSTS = ["*"]/' techynotes/settings.py
CMD ["/usr/bin/python3", "manage.py", "runserver","0.0.0.0:9000"]
